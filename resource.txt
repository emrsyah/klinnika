export const ObservableFromRef = (
  ref: CollectionReference<DocumentData, DocumentData>
) => {
  return new Observable((subscriber) => {
    const unsubscribe = onSnapshot(ref, (snapshot) =>
      subscriber.next(snapshot)
    );
    return () => unsubscribe();
  });
};
